以下是基于 ClickZetta Lakehouse 构建的完整数据管道，包含 ODS 自动导入、DWD 清洗和 DWS 聚合三层，每小时刷新一次。

================================================================================
第一步：ODS 层 —— 从 OSS Volume 自动导入原始日志
================================================================================

-- 1. 创建 ODS 原始表（外部表映射 OSS Volume）
CREATE EXTERNAL TABLE IF NOT EXISTS ods_app_log (
    log_id      STRING,
    user_id     STRING,
    action      STRING,
    duration_ms BIGINT,
    log_time    TIMESTAMP
)
STORED AS PARQUET
LOCATION 'volume://my_oss_vol/logs/app/';

-- 说明：
-- 使用 EXTERNAL TABLE + LOCATION 指向 Volume 路径，ClickZetta 会自动扫描该路径下所有 Parquet 文件。
-- 每次查询时自动读取最新上传的文件，无需手动导入。

-- 2. 如果需要将数据落地到内部 ODS 表（推荐，便于后续处理）：
CREATE TABLE IF NOT EXISTS ods_app_log_internal (
    log_id      STRING,
    user_id     STRING,
    action      STRING,
    duration_ms BIGINT,
    log_time    TIMESTAMP,
    etl_date    DATE,
    etl_hour    INT
)
PARTITIONED BY (etl_date, etl_hour);

-- 每小时增量导入任务（通过调度器每小时执行）：
INSERT INTO ods_app_log_internal
    PARTITION (etl_date = CURRENT_DATE, etl_hour = HOUR(CURRENT_TIMESTAMP))
SELECT
    log_id,
    user_id,
    action,
    duration_ms,
    log_time,
    CAST(log_time AS DATE)  AS etl_date,
    HOUR(log_time)          AS etl_hour
FROM ods_app_log
WHERE CAST(log_time AS DATE) = CURRENT_DATE
  AND HOUR(log_time) = HOUR(CURRENT_TIMESTAMP) - 1;  -- 导入上一小时的数据


================================================================================
第二步：DWD 层 —— 数据清洗（过滤无效 duration_ms）
================================================================================

-- 创建 DWD 清洗表
CREATE TABLE IF NOT EXISTS dwd_app_log_clean (
    log_id      STRING,
    user_id     STRING,
    action      STRING,
    duration_ms BIGINT,
    log_time    TIMESTAMP,
    log_date    DATE,
    log_hour    INT
)
PARTITIONED BY (log_date, log_hour);

-- 每小时清洗任务（调度器每小时执行，处理上一小时分区）：
INSERT INTO dwd_app_log_clean
    PARTITION (log_date, log_hour)
SELECT
    log_id,
    user_id,
    action,
    duration_ms,
    log_time,
    CAST(log_time AS DATE) AS log_date,
    HOUR(log_time)         AS log_hour
FROM ods_app_log_internal
WHERE etl_date = CURRENT_DATE
  AND etl_hour = HOUR(CURRENT_TIMESTAMP) - 1
  AND duration_ms > 0          -- 核心清洗规则：过滤 duration_ms <= 0 的无效记录
  AND log_id   IS NOT NULL
  AND user_id  IS NOT NULL
  AND action   IS NOT NULL
  AND log_time IS NOT NULL;


================================================================================
第三步：DWS 层 —— 按天统计每个 action 的平均耗时和次数
================================================================================

-- 创建 DWS 汇总表
CREATE TABLE IF NOT EXISTS dws_action_daily_stats (
    log_date        DATE,
    action          STRING,
    total_count     BIGINT,
    avg_duration_ms DOUBLE,
    max_duration_ms BIGINT,
    min_duration_ms BIGINT,
    updated_at      TIMESTAMP
)
PARTITIONED BY (log_date);

-- 每小时刷新当天汇总（覆盖写当天分区，保证数据最新）：
INSERT OVERWRITE INTO dws_action_daily_stats
    PARTITION (log_date = CURRENT_DATE)
SELECT
    CURRENT_DATE                    AS log_date,
    action,
    COUNT(*)                        AS total_count,
    AVG(duration_ms)                AS avg_duration_ms,
    MAX(duration_ms)                AS max_duration_ms,
    MIN(duration_ms)                AS min_duration_ms,
    CURRENT_TIMESTAMP               AS updated_at
FROM dwd_app_log_clean
WHERE log_date = CURRENT_DATE
GROUP BY action;


================================================================================
第四步：调度配置（ClickZetta Task / 外部调度器）
================================================================================

-- 使用 ClickZetta Task 实现每小时自动调度：

-- Task 1：ODS 导入
CREATE TASK task_ods_load
    SCHEDULE = 'EVERY 1 HOUR'
    AS
INSERT INTO ods_app_log_internal
    PARTITION (etl_date = CURRENT_DATE, etl_hour = HOUR(CURRENT_TIMESTAMP))
SELECT
    log_id,
    user_id,
    action,
    duration_ms,
    log_time,
    CAST(log_time AS DATE),
    HOUR(log_time)
FROM ods_app_log
WHERE CAST(log_time AS DATE) = CURRENT_DATE
  AND HOUR(log_time) = HOUR(CURRENT_TIMESTAMP) - 1;

-- Task 2：DWD 清洗（依赖 ODS 完成后执行）
CREATE TASK task_dwd_clean
    SCHEDULE = 'EVERY 1 HOUR'
    AFTER task_ods_load
    AS
INSERT INTO dwd_app_log_clean
    PARTITION (log_date, log_hour)
SELECT
    log_id,
    user_id,
    action,
    duration_ms,
    log_time,
    CAST(log_time AS DATE),
    HOUR(log_time)
FROM ods_app_log_internal
WHERE etl_date = CURRENT_DATE
  AND etl_hour = HOUR(CURRENT_TIMESTAMP) - 1
  AND duration_ms > 0
  AND log_id IS NOT NULL
  AND user_id IS NOT NULL
  AND action IS NOT NULL
  AND log_time IS NOT NULL;

-- Task 3：DWS 聚合（依赖 DWD 完成后执行）
CREATE TASK task_dws_aggregate
    SCHEDULE = 'EVERY 1 HOUR'
    AFTER task_dwd_clean
    AS
INSERT OVERWRITE INTO dws_action_daily_stats
    PARTITION (log_date = CURRENT_DATE)
SELECT
    CURRENT_DATE,
    action,
    COUNT(*),
    AVG(duration_ms),
    MAX(duration_ms),
    MIN(duration_ms),
    CURRENT_TIMESTAMP
FROM dwd_app_log_clean
WHERE log_date = CURRENT_DATE
GROUP BY action;

-- 启动所有 Task：
ALTER TASK task_ods_load RESUME;
ALTER TASK task_dwd_clean RESUME;
ALTER TASK task_dws_aggregate RESUME;


================================================================================
第五步：验证查询
================================================================================

-- 查看今天各 action 的统计结果：
SELECT
    action,
    total_count,
    ROUND(avg_duration_ms, 2) AS avg_duration_ms,
    max_duration_ms,
    min_duration_ms,
    updated_at
FROM dws_action_daily_stats
WHERE log_date = CURRENT_DATE
ORDER BY total_count DESC;

-- 查看最近 7 天趋势：
SELECT
    log_date,
    action,
    total_count,
    ROUND(avg_duration_ms, 2) AS avg_duration_ms
FROM dws_action_daily_stats
WHERE log_date >= CURRENT_DATE - INTERVAL 7 DAY
ORDER BY log_date DESC, total_count DESC;


================================================================================
架构总结
================================================================================

数据流向：
OSS (my_oss_vol/logs/app/*.parquet)
  └─> ods_app_log (外部表，实时映射)
        └─> ods_app_log_internal (内部分区表，按小时落地)
              └─> dwd_app_log_clean (清洗：过滤 duration_ms <= 0)
                    └─> dws_action_daily_stats (按天聚合：action 维度)

调度策略：
- 每小时触发一次，处理上一小时的增量数据
- ODS → DWD → DWS 串行依赖执行，保证数据一致性
- DWS 使用 INSERT OVERWRITE 覆盖当天分区，每次刷新都是当天最新全量

关键设计决策：
1. 外部表 (ods_app_log) 直接映射 OSS Volume，无需额外导入步骤，新文件自动可见
2. 内部 ODS 表按 (etl_date, etl_hour) 分区，支持高效增量处理和数据回溯
3. DWD 清洗规则：duration_ms > 0，同时过滤关键字段的 NULL 值
4. DWS 按 log_date 分区，每小时覆盖写当天数据，查询性能好且数据始终最新
