针对 ClickZetta Lakehouse 构建近实时数仓管道（ODS -> DWD -> DWS），延迟目标 1 分钟以内。

================================================================================
架构说明
================================================================================

ODS 层：ods.orders（已有，持续接收 INSERT/UPDATE）
DWD 层：dwd.orders_latest（保留每个 order_id 的最新状态）
DWS 层：dws.orders_daily_status（按天、按 status 统计订单数和金额）

核心思路：
- DWD 层使用 MERGE INTO 做 upsert，以 order_id 为主键，保留最新 updated_at 的记录
- DWS 层基于 DWD 层做聚合，使用 INSERT OVERWRITE 按分区刷新当天数据
- 通过 ClickZetta 的 Task（定时任务）每分钟调度，实现近实时

================================================================================
Step 1：创建 DWD 层表
================================================================================

-- DWD 层：每个 order_id 只保留最新一条记录
CREATE TABLE IF NOT EXISTS dwd.orders_latest (
    order_id     BIGINT        NOT NULL,
    customer_id  BIGINT,
    amount       DECIMAL(18,2),
    status       VARCHAR(50),
    updated_at   TIMESTAMP,
    dwd_load_ts  TIMESTAMP     -- 写入 DWD 的时间戳
)
USING DELTA
PRIMARY KEY (order_id);

================================================================================
Step 2：创建 DWS 层表
================================================================================

-- DWS 层：按日期 + status 聚合
CREATE TABLE IF NOT EXISTS dws.orders_daily_status (
    stat_date    DATE          NOT NULL,
    status       VARCHAR(50)   NOT NULL,
    order_cnt    BIGINT,
    total_amount DECIMAL(18,2),
    dws_load_ts  TIMESTAMP
)
USING DELTA
PRIMARY KEY (stat_date, status);

================================================================================
Step 3：ODS -> DWD 增量同步任务
================================================================================

-- 使用 MERGE INTO 将 ODS 中有变更的数据 upsert 到 DWD
-- 每次只处理最近 2 分钟内有更新的记录（覆盖调度间隔，留有余量）

MERGE INTO dwd.orders_latest AS target
USING (
    -- 从 ODS 取最近 2 分钟内更新的记录，若同一 order_id 有多条取最新的
    SELECT
        order_id,
        customer_id,
        amount,
        status,
        updated_at,
        CURRENT_TIMESTAMP() AS dwd_load_ts
    FROM (
        SELECT
            order_id,
            customer_id,
            amount,
            status,
            updated_at,
            ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY updated_at DESC) AS rn
        FROM ods.orders
        WHERE updated_at >= CURRENT_TIMESTAMP() - INTERVAL 2 MINUTES
    ) t
    WHERE rn = 1
) AS source
ON target.order_id = source.order_id
WHEN MATCHED AND source.updated_at > target.updated_at THEN
    UPDATE SET
        customer_id  = source.customer_id,
        amount       = source.amount,
        status       = source.status,
        updated_at   = source.updated_at,
        dwd_load_ts  = source.dwd_load_ts
WHEN NOT MATCHED THEN
    INSERT (order_id, customer_id, amount, status, updated_at, dwd_load_ts)
    VALUES (source.order_id, source.customer_id, source.amount, source.status, source.updated_at, source.dwd_load_ts);

================================================================================
Step 4：DWD -> DWS 聚合刷新任务
================================================================================

-- 刷新今天和昨天的聚合数据（防止跨天边界遗漏）
-- 使用 MERGE INTO 做 upsert，避免重复写入

MERGE INTO dws.orders_daily_status AS target
USING (
    SELECT
        CAST(updated_at AS DATE)  AS stat_date,
        status,
        COUNT(*)                  AS order_cnt,
        SUM(amount)               AS total_amount,
        CURRENT_TIMESTAMP()       AS dws_load_ts
    FROM dwd.orders_latest
    WHERE CAST(updated_at AS DATE) >= CURRENT_DATE() - INTERVAL 1 DAY
    GROUP BY CAST(updated_at AS DATE), status
) AS source
ON target.stat_date = source.stat_date
   AND target.status = source.status
WHEN MATCHED THEN
    UPDATE SET
        order_cnt    = source.order_cnt,
        total_amount = source.total_amount,
        dws_load_ts  = source.dws_load_ts
WHEN NOT MATCHED THEN
    INSERT (stat_date, status, order_cnt, total_amount, dws_load_ts)
    VALUES (source.stat_date, source.status, source.order_cnt, source.total_amount, source.dws_load_ts);

================================================================================
Step 5：创建调度 Task（每分钟执行）
================================================================================

-- Task 1：ODS -> DWD，每分钟执行
CREATE TASK IF NOT EXISTS task_ods_to_dwd
    SCHEDULE = '1 MINUTE'
    COMMENT = 'ODS orders upsert to DWD orders_latest'
AS
MERGE INTO dwd.orders_latest AS target
USING (
    SELECT
        order_id,
        customer_id,
        amount,
        status,
        updated_at,
        CURRENT_TIMESTAMP() AS dwd_load_ts
    FROM (
        SELECT
            order_id,
            customer_id,
            amount,
            status,
            updated_at,
            ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY updated_at DESC) AS rn
        FROM ods.orders
        WHERE updated_at >= CURRENT_TIMESTAMP() - INTERVAL 2 MINUTES
    ) t
    WHERE rn = 1
) AS source
ON target.order_id = source.order_id
WHEN MATCHED AND source.updated_at > target.updated_at THEN
    UPDATE SET
        customer_id  = source.customer_id,
        amount       = source.amount,
        status       = source.status,
        updated_at   = source.updated_at,
        dwd_load_ts  = source.dwd_load_ts
WHEN NOT MATCHED THEN
    INSERT (order_id, customer_id, amount, status, updated_at, dwd_load_ts)
    VALUES (source.order_id, source.customer_id, source.amount, source.status, source.updated_at, source.dwd_load_ts);

-- Task 2：DWD -> DWS，每分钟执行（依赖 task_ods_to_dwd 完成后触发更优，此处简化为独立调度）
CREATE TASK IF NOT EXISTS task_dwd_to_dws
    SCHEDULE = '1 MINUTE'
    COMMENT = 'DWD orders_latest aggregate to DWS orders_daily_status'
AS
MERGE INTO dws.orders_daily_status AS target
USING (
    SELECT
        CAST(updated_at AS DATE)  AS stat_date,
        status,
        COUNT(*)                  AS order_cnt,
        SUM(amount)               AS total_amount,
        CURRENT_TIMESTAMP()       AS dws_load_ts
    FROM dwd.orders_latest
    WHERE CAST(updated_at AS DATE) >= CURRENT_DATE() - INTERVAL 1 DAY
    GROUP BY CAST(updated_at AS DATE), status
) AS source
ON target.stat_date = source.stat_date
   AND target.status = source.status
WHEN MATCHED THEN
    UPDATE SET
        order_cnt    = source.order_cnt,
        total_amount = source.total_amount,
        dws_load_ts  = source.dws_load_ts
WHEN NOT MATCHED THEN
    INSERT (stat_date, status, order_cnt, total_amount, dws_load_ts)
    VALUES (source.stat_date, source.status, source.order_cnt, source.total_amount, source.dws_load_ts);

-- 启动 Task
ALTER TASK task_ods_to_dwd RESUME;
ALTER TASK task_dwd_to_dws RESUME;

================================================================================
Step 6：验证查询
================================================================================

-- 查看 DWD 最新数据
SELECT * FROM dwd.orders_latest ORDER BY updated_at DESC LIMIT 20;

-- 查看 DWS 今日各状态汇总
SELECT stat_date, status, order_cnt, total_amount
FROM dws.orders_daily_status
WHERE stat_date = CURRENT_DATE()
ORDER BY status;

================================================================================
关键设计说明
================================================================================

1. 延迟控制：
   - Task 每 1 分钟触发一次，ODS -> DWD -> DWS 两跳串行最坏情况约 2 分钟
   - 若需严格 1 分钟内，可将 task_dwd_to_dws 设置为 task_ods_to_dwd 的下游依赖任务（AFTER task_ods_to_dwd），两个任务在同一分钟内顺序完成

2. MERGE INTO 的幂等性：
   - DWD 层通过 updated_at 比较，只有更新的数据才会覆盖，保证幂等
   - DWS 层全量重算当天+昨天分区，天然幂等

3. ODS 窗口选择 2 分钟：
   - 调度间隔 1 分钟，取 2 分钟窗口可容忍轻微的调度抖动，避免漏数据
   - MERGE INTO 的 updated_at 比较保证重复处理不会产生脏数据

4. DWS 刷新范围：
   - 刷新今天和昨天，处理跨天边界的订单状态变更（如昨天下单今天更新状态）

5. 表格式使用 DELTA：
   - 支持 MERGE INTO upsert 操作，适合 CDC 场景
