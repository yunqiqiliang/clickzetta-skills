以下是完整的三层数据管道 SQL，覆盖从 Kafka 接入到 DWS 聚合的端到端流程。

================================================================================
数据管道设计说明
================================================================================

数据来源：Kafka（broker: kafka.prod.com:9092，topic: user_events，JSON 格式）
字段：user_id、event_type、page_id、ts（时间戳）
分层：ODS 原始层 → DWD 清洗层 → DWS 聚合层
刷新策略：DWD 每 2 分钟刷新，DWS 跟随 DWD（DOWNSTREAM）

================================================================================
Step 0：创建 Schema
================================================================================

CREATE SCHEMA IF NOT EXISTS ods;
CREATE SCHEMA IF NOT EXISTS dwd;
CREATE SCHEMA IF NOT EXISTS dws;

================================================================================
Step 1：ODS 层 — 建目标表
================================================================================

-- ODS 原始表，存储从 Kafka 摄入的原始用户行为数据
CREATE TABLE IF NOT EXISTS ods.user_events (
  user_id     STRING,
  event_type  STRING,
  page_id     STRING,
  ts          TIMESTAMP
);

================================================================================
Step 2：ODS 层 — 创建 Pipe 持续从 Kafka 摄入数据
================================================================================

-- Pipe 创建后自动启动，持续将 Kafka 消息写入 ods.user_events
CREATE OR REPLACE PIPE kafka_user_events_pipe
  COMMENT '从 Kafka topic user_events 持续摄入用户行为数据到 ODS 层'
AS
INSERT INTO ods.user_events (user_id, event_type, page_id, ts)
SELECT
  $1:user_id::STRING,
  $1:event_type::STRING,
  $1:page_id::STRING,
  $1:ts::TIMESTAMP
FROM TABLE(
  READ_KAFKA(
    KAFKA_BROKER      => 'kafka.prod.com:9092',
    KAFKA_TOPIC       => 'user_events',
    KAFKA_GROUP_ID    => 'lakehouse_user_events_ingest',
    KAFKA_OFFSET      => 'latest',
    KAFKA_DATA_FORMAT => 'json'
  )
);

================================================================================
Step 3：DWD 层 — 清洗动态表（每 2 分钟刷新）
================================================================================

-- 过滤无效 event_type（NULL 或空字符串），保留有效的用户行为记录
-- 同时提取日期字段 dt，便于下游按天聚合
CREATE OR REPLACE DYNAMIC TABLE dwd.user_events_clean
  TARGET_LAG = '2 minutes'
  VCLUSTER = default_ap
  COMMENT 'DWD 清洗层：过滤无效 event_type，每 2 分钟增量刷新'
AS
SELECT
  user_id,
  event_type,
  page_id,
  ts,
  DATE(ts) AS dt
FROM ods.user_events
WHERE event_type IS NOT NULL
  AND event_type <> '';

================================================================================
Step 4：DWS 层 — 聚合动态表（跟随 DWD 刷新）
================================================================================

-- 按天统计每个 page 的 UV（独立访客数）和 PV（页面浏览量）
-- TARGET_LAG = DOWNSTREAM 表示由上游 dwd.user_events_clean 驱动刷新
CREATE OR REPLACE DYNAMIC TABLE dws.page_uv_pv_daily
  TARGET_LAG = DOWNSTREAM
  VCLUSTER = default_ap
  COMMENT 'DWS 聚合层：按天统计每个 page 的 UV 和 PV，跟随 DWD 自动刷新'
AS
SELECT
  dt,
  page_id,
  COUNT(DISTINCT user_id) AS uv,
  COUNT(*)                AS pv
FROM dwd.user_events_clean
GROUP BY dt, page_id;

================================================================================
Step 5：验证命令
================================================================================

-- 验证 Pipe 是否正常运行
SHOW PIPES;
DESC PIPE kafka_user_events_pipe;

-- 验证动态表是否创建成功
SHOW TABLES WHERE is_dynamic = true;

-- 查看 DWD 层刷新历史（确认每 2 分钟触发）
SHOW DYNAMIC TABLE REFRESH HISTORY dwd.user_events_clean LIMIT 10;

-- 查看 DWS 层刷新历史（确认跟随 DWD 触发）
SHOW DYNAMIC TABLE REFRESH HISTORY dws.page_uv_pv_daily LIMIT 10;

-- 抽查 ODS 原始数据
SELECT * FROM ods.user_events LIMIT 20;

-- 抽查 DWD 清洗结果（确认无效 event_type 已过滤）
SELECT DISTINCT event_type FROM dwd.user_events_clean LIMIT 20;

-- 查看 DWS 聚合结果
SELECT * FROM dws.page_uv_pv_daily ORDER BY dt DESC, pv DESC LIMIT 20;

================================================================================
Step 6：运维操作（按需执行）
================================================================================

-- 暂停整条管道（如集群维护）
ALTER PIPE kafka_user_events_pipe PAUSE;
ALTER DYNAMIC TABLE dwd.user_events_clean SUSPEND;
ALTER DYNAMIC TABLE dws.page_uv_pv_daily SUSPEND;

-- 恢复整条管道
ALTER PIPE kafka_user_events_pipe RESUME;
ALTER DYNAMIC TABLE dwd.user_events_clean RESUME;
ALTER DYNAMIC TABLE dws.page_uv_pv_daily RESUME;

-- 排查 DWD 刷新失败
SHOW DYNAMIC TABLE REFRESH HISTORY dwd.user_events_clean LIMIT 20;

-- 排查 DWS 刷新失败
SHOW DYNAMIC TABLE REFRESH HISTORY dws.page_uv_pv_daily LIMIT 20;

================================================================================
架构说明
================================================================================

Kafka (kafka.prod.com:9092 / user_events)
  |
  | [Pipe: kafka_user_events_pipe]
  | 持续摄入，READ_KAFKA，JSON 解析
  v
ods.user_events（原始表）
  |
  | [Dynamic Table: dwd.user_events_clean]
  | TARGET_LAG = '2 minutes'
  | 过滤 event_type IS NULL 或空字符串
  v
dwd.user_events_clean（清洗层）
  |
  | [Dynamic Table: dws.page_uv_pv_daily]
  | TARGET_LAG = DOWNSTREAM（跟随 DWD）
  | GROUP BY dt, page_id → UV + PV
  v
dws.page_uv_pv_daily（聚合层）

关键设计决策：
1. ODS 层使用普通表 + Pipe，Pipe 创建后自动持续摄入，无需调度
2. DWD 层 TARGET_LAG = '2 minutes'，每 2 分钟增量刷新一次
3. DWS 层 TARGET_LAG = DOWNSTREAM，DWD 刷新完成后立即触发 DWS 刷新
4. UV 使用 COUNT(DISTINCT user_id)，PV 使用 COUNT(*)
5. 无效 event_type 过滤条件：IS NOT NULL AND <> ''（覆盖 NULL 和空字符串两种情况）
