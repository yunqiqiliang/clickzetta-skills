根据你的需求，以下是完整的端到端数据管道 SQL：

数据来源：OSS Volume（my_oss_vol，路径 /logs/app/，Parquet 格式）
字段：log_id、user_id、action、duration_ms、log_time
分层：ODS（原始导入）→ DWD（清洗）→ DWS（聚合）
刷新频率：每小时一次

================================================================================
-- Step 1: 创建 Schema
================================================================================

CREATE SCHEMA IF NOT EXISTS ods;
CREATE SCHEMA IF NOT EXISTS dwd;
CREATE SCHEMA IF NOT EXISTS dws;

================================================================================
-- Step 2: 创建 ODS 层目标表（接收 Pipe 导入的原始数据）
================================================================================

CREATE TABLE IF NOT EXISTS ods.app_logs (
  log_id      STRING,
  user_id     STRING,
  action      STRING,
  duration_ms BIGINT,
  log_time    TIMESTAMP
);

================================================================================
-- Step 3: 创建 Pipe，从 OSS 持续导入 Parquet 文件到 ODS
================================================================================

CREATE OR REPLACE PIPE oss_app_logs_pipe
  AUTO_INGEST = TRUE
  COMMENT '从 OSS my_oss_vol/logs/app/ 持续导入应用日志 Parquet 文件'
AS
COPY INTO ods.app_logs
FROM '@my_oss_vol/logs/app/'
FILE_FORMAT = ( TYPE = 'parquet' )
PATTERN = '.*\.parquet';

================================================================================
-- Step 4: 创建 DWD 层动态表（清洗：过滤 duration_ms <= 0，每小时刷新）
================================================================================

CREATE OR REPLACE DYNAMIC TABLE dwd.app_logs_clean
  TARGET_LAG = '1 hours'
  VCLUSTER = default_ap
  COMMENT 'DWD 层：过滤无效 duration_ms，补充日期分区字段'
AS
SELECT
  log_id,
  user_id,
  action,
  duration_ms,
  log_time,
  DATE(log_time) AS dt
FROM ods.app_logs
WHERE duration_ms > 0;

================================================================================
-- Step 5: 创建 DWS 层动态表（按天统计每个 action 的平均耗时和次数，跟随上游刷新）
================================================================================

CREATE OR REPLACE DYNAMIC TABLE dws.app_action_daily_stats
  TARGET_LAG = DOWNSTREAM
  VCLUSTER = default_ap
  COMMENT 'DWS 层：按天统计每个 action 的平均耗时和请求次数'
AS
SELECT
  dt,
  action,
  COUNT(*)                        AS request_cnt,
  AVG(duration_ms)                AS avg_duration_ms
FROM dwd.app_logs_clean
GROUP BY dt, action;

================================================================================
-- Step 6: 验证命令
================================================================================

-- 查看 Pipe 状态
SHOW PIPES;
DESC PIPE oss_app_logs_pipe;

-- 查看动态表列表
SHOW TABLES IN dwd WHERE is_dynamic = true;
SHOW TABLES IN dws WHERE is_dynamic = true;

-- 查看 DWD 动态表刷新历史
SHOW DYNAMIC TABLE REFRESH HISTORY dwd.app_logs_clean LIMIT 5;

-- 查看 DWS 动态表刷新历史
SHOW DYNAMIC TABLE REFRESH HISTORY dws.app_action_daily_stats LIMIT 5;

-- 抽查 DWD 数据
SELECT COUNT(*) FROM dwd.app_logs_clean;

-- 抽查 DWS 数据
SELECT * FROM dws.app_action_daily_stats ORDER BY dt DESC, request_cnt DESC LIMIT 20;

================================================================================
-- Step 7: 运维操作（按需使用）
================================================================================

-- 暂停整条管道（如集群维护）
ALTER PIPE oss_app_logs_pipe PAUSE;
ALTER DYNAMIC TABLE dwd.app_logs_clean SUSPEND;
ALTER DYNAMIC TABLE dws.app_action_daily_stats SUSPEND;

-- 恢复整条管道
ALTER PIPE oss_app_logs_pipe RESUME;
ALTER DYNAMIC TABLE dwd.app_logs_clean RESUME;
ALTER DYNAMIC TABLE dws.app_action_daily_stats RESUME;

================================================================================
-- 管道说明
================================================================================

数据流向：
  OSS（my_oss_vol/logs/app/*.parquet）
    → [Pipe: oss_app_logs_pipe]
    → ods.app_logs（原始表）
    → [Dynamic Table: dwd.app_logs_clean，TARGET_LAG = '1 hours']
      过滤 duration_ms <= 0，补充 dt 字段
    → [Dynamic Table: dws.app_action_daily_stats，TARGET_LAG = DOWNSTREAM]
      按 dt + action 分组，统计 request_cnt 和 avg_duration_ms

刷新机制：
  - Pipe 检测到 OSS 新文件后自动触发 COPY INTO，无需调度
  - DWD 动态表每小时增量刷新一次（TARGET_LAG = '1 hours'）
  - DWS 动态表在 DWD 刷新完成后自动跟随刷新（TARGET_LAG = DOWNSTREAM）
  - 整体端到端延迟约 1 小时，满足"每小时刷新一次"的需求
